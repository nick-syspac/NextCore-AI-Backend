Evaluation Process

def evaluate(context: EvaluationContext) -> EvaluationResult:
    # Step 1: Load ruleset
    ruleset = Ruleset.objects.get(id=ruleset_id)
    artifacts = ruleset.artifacts.all()
    
    # Step 2: For each artifact
    results = []
    for artifact in artifacts:
        # Parse JSONLogic
        rule = json.loads(artifact.blob)
        
        # Evaluate with context
        result = jsonlogic_evaluate(rule, context.to_dict())
        results.append(result)
    
    # Step 3: Aggregate results
    if all(results):
        outcome = "eligible"
    elif any(r.get("requires_review") for r in results):
        outcome = "review"
    else:
        outcome = "ineligible"
    
    # Step 4: Generate explanation
    reasons = extract_reasons(results)
    clause_refs = extract_clauses(results)
    
    return EvaluationResult(
        outcome=outcome,
        reasons=reasons,
        clause_refs=clause_refs,
        explanation=generate_explanation(outcome, reasons)
    )

Connector Framework

Base Connector Architecture

class BaseConnector:
    """Abstract base for external API connectors"""
    
    def __init__(self, timeout=30, cache_ttl=3600):
        self.timeout = timeout
        self.cache_ttl = cache_ttl
    
    def lookup(self, data: dict) -> dict:
        # Check cache first
        cache_key = self.generate_cache_key(data)
        cached = redis.get(cache_key)
        if cached:
            return json.loads(cached)
        
        # Call external API
        result = self._call_api(data)
        
        # Cache result
        redis.setex(cache_key, self.cache_ttl, json.dumps(result))
        
        # Store in database
        ExternalLookup.objects.create(
            request=request,
            provider=self.provider_name,
            lookup_data=data,
            result_data=result,
            cached=False
        )
        
        return result
    
    def _call_api(self, data: dict) -> dict:
        raise NotImplementedError

@shared_task
def enqueue_external_lookups(request_id):
    """Determine which lookups are needed"""
    request = EligibilityRequest.objects.get(id=request_id)
    
    # Always lookup USI and postcode
    tasks = [
        perform_external_lookup.s(request_id, "USI", {...}),
        perform_external_lookup.s(request_id, "Postcode", {...})
    ]
    
    # Conditional lookups
    if request.student_data.get("has_concession_card"):
        tasks.append(perform_external_lookup.s(request_id, "Concession", {...}))
    
    if request.student_data.get("citizenship") == "Visa":
        tasks.append(perform_external_lookup.s(request_id, "Visa", {...}))
    
    # Execute in parallel
    group(tasks).apply_async()


@shared_task(bind=True, max_retries=3)
def perform_external_lookup(self, request_id, provider, data):
    """Execute single external lookup with retry"""
    try:
        connector = ConnectorFactory.create(provider)
        result = connector.lookup(data)
        return {"provider": provider, "result": result, "success": True}
    except Exception as exc:
        self.retry(exc=exc, countdown=30)


@shared_task
def evaluate_eligibility(request_id, ruleset_id):
    """Run rules engine evaluation"""
    request = EligibilityRequest.objects.get(id=request_id)
    ruleset = Ruleset.objects.get(id=ruleset_id)
    
    # Build context
    context = EvaluationContext(
        input_data=request.get_input_snapshot(),
        lookups=request.get_lookup_results(),
        reference_tables=ruleset.get_reference_tables()
    )
    
    # Evaluate
    engine = RulesEngine()
    result = engine.evaluate(context, ruleset)
    
    # Store decision
    decision = EligibilityDecision.objects.create(
        request=request,
        outcome=result.outcome,
        reasons=result.reasons,
        clause_refs=result.clause_refs,
        decision_data=result.to_dict(),
        ruleset=ruleset
    )
    
    # Update request status
    request.status = "evaluated"
    request.save()
    
    # Queue webhook
    deliver_webhook.delay(request_id, "decision.finalized")

Webhook Delivery System

Delivery Flow

@shared_task(bind=True, max_retries=5)
def deliver_webhook(self, request_id, event_type):
    """Deliver webhook with HMAC signature"""
    request = EligibilityRequest.objects.get(id=request_id)
    
    # Get active webhooks for event
    endpoints = WebhookEndpoint.objects.filter(
        tenant=request.tenant,
        active=True,
        events__contains=[event_type]
    )
    
    for endpoint in endpoints:
        # Build payload
        payload = {
            "event_type": event_type,
            "request_id": request_id,
            "person_id": request.person_id,
            "course_id": request.course_id,
            "jurisdiction": request.jurisdiction_code,
            "timestamp": timezone.now().isoformat()
        }
        
        # Generate HMAC signature
        signature = hmac.new(
            endpoint.secret.encode(),
            json.dumps(payload).encode(),
            hashlib.sha256
        ).hexdigest()
        
        # Send HTTP POST
        response = requests.post(
            endpoint.url,
            json=payload,
            headers={
                "X-Webhook-Signature": signature,
                "X-Webhook-Event": event_type,
                "Content-Type": "application/json"
            },
            timeout=30
        )
        
        # Log delivery
        WebhookDelivery.objects.create(
            endpoint=endpoint,
            request=request,
            event_type=event_type,
            payload=payload,
            response_status=response.status_code,
            success=(200 <= response.status_code < 300)
        )

Security Architecture
