üìö Step 1: Seed ASQA Standards

Before comparing policies, you need to populate the ASQA standards database. Here's how to seed Standard 1.1 (Training and Assessment Strategies) as an example:

# save as: seed_asqa_standards.py from policy_comparator.models import ASQAStandard, ASQAClause # Create Standard 1.1 standard = ASQAStandard.objects.create( standard_number="1.1", title="Training and Assessment Strategies", description="RTOs must have training and assessment strategies that meet the requirements of the training package", standard_type="training_assessment", full_text=""" The RTO's training and assessment strategies and practices, including the amount of training they provide, are consistent with the requirements of training packages and VET accredited courses and enable each learner to meet the requirements for each unit of competency or module in which they are enrolled. """, requirements=[ "Develop training and assessment strategies for each training product", "Ensure strategies are consistent with training package requirements", "Provide sufficient training to meet competency requirements", "Adapt strategies to meet learner needs", "Document and implement strategies effectively" ], is_active=True, version="2015" ) # Create clauses for Standard 1.1 clauses_data = [ { "clause_number": "1.1.1", "title": "Training and Assessment Strategy Requirements", "clause_text": "The RTO must develop and implement a training and assessment strategy for each training product on its scope of registration.", "keywords": ["training and assessment strategy", "training product", "scope of registration", "develop", "implement"], "compliance_level": "critical", "evidence_required": ["TAS documents", "Implementation evidence", "Scope of registration"] }, { "clause_number": "1.1.2", "title": "Training Package Consistency", "clause_text": "Training and assessment must be consistent with the requirements specified in the training package.", "keywords": ["training package requirements", "consistency", "units of competency", "assessment requirements"], "compliance_level": "critical", "evidence_required": ["Training materials", "Assessment tools", "Mapping documents"] }, { "clause_number": "1.1.3", "title": "Sufficient Training Duration", "clause_text": "The amount and nature of training provided must be sufficient to enable learners to meet the requirements of each unit of competency.", "keywords": ["training duration", "sufficient training", "learner needs", "competency requirements"], "compliance_level": "essential", "evidence_required": ["Training schedules", "Delivery plans", "Learner contact hours"] } ] for clause_data in clauses_data: ASQAClause.objects.create( standard=standard, **clause_data ) print(f"‚úÖ Created Standard {standard.standard_number} with {len(clauses_data)} clauses")

üìù Step 2: Create a Policy

Create an RTO policy to compare against ASQA standards:

import requests from datetime import date # API endpoint (replace with your tenant slug) url = "https://api.nextcore.ai/v1/{tenant_slug}/policy-comparator/policies/" # Authentication header headers = { "Authorization": "Bearer YOUR_AUTH_TOKEN", "Content-Type": "application/json" } # Policy data policy_data = { "policy_number": "TAS-001", "title": "Training and Assessment Strategy Policy", "description": "Policy outlining TAS development and implementation requirements", "policy_type": "training_delivery", "content": """ Training and Assessment Strategy Policy 1. Purpose Our RTO develops comprehensive training and assessment strategies for each qualification and unit of competency on our scope of registration. These strategies ensure consistent delivery of quality training that meets the requirements of training packages. 2. Strategy Development - TAS documents are created for all training products - Strategies are reviewed annually and updated as needed - Training package requirements are incorporated into all strategies - Industry consultation informs strategy development 3. Training Delivery - Sufficient training duration to meet competency requirements - Flexible delivery modes to accommodate learner needs - Resources and facilities appropriate for training delivery - Regular monitoring of training effectiveness 4. Assessment Practices - Assessment tools aligned with unit requirements - Valid and reliable assessment methods - Assessment conducted by qualified assessors - Reasonable adjustment provided where needed 5. Documentation and Review - All TAS documents maintained and accessible - Strategies reviewed after each delivery - Continuous improvement based on feedback """, "version": "1.0", "status": "approved", "effective_date": date.today().isoformat() } # Create policy response = requests.post(url, json=policy_data, headers=headers) if response.status_code == 201: policy = response.json() print(f"‚úÖ Policy created with ID: {policy['id']}") print(f" Policy Number: {policy['policy_number']}") print(f" Title: {policy['title']}") else: print(f"‚ùå Error: {response.status_code} - {response.text}")

üîç Step 3: Run Comparison

Now compare the policy against ASQA standards. The comparison will analyze the policy text using NLP and identify compliance gaps:

import requests # Comparison endpoint url = f"https://api.nextcore.ai/v1/{{tenant_slug}}/policy-comparator/policies/{policy['id']}/compare/" headers = { "Authorization": "Bearer YOUR_AUTH_TOKEN", "Content-Type": "application/json" } # Comparison request comparison_data = { "policy_id": policy['id'], "standard_ids": [1], # Compare against Standard 1 only, or omit for all "session_name": "Initial TAS Policy Comparison", "use_nlp": True } # Start comparison response = requests.post(url, json=comparison_data, headers=headers) if response.status_code == 200: result = response.json() print("‚úÖ Comparison Completed!") print(f" Session ID: {result['session']['id']}") print(f" Processing Time: {result['processing_time']:.2f} seconds") print(f"\nüìä Results Summary:") print(f" Total Clauses Checked: {result['results_summary']['total_checked']}") print(f" Compliant: {result['results_summary']['compliant']}") print(f" Partial Matches: {result['results_summary']['partial_match']}") print(f" Gaps Found: {result['results_summary']['gaps']}") print(f" Overall Compliance Score: {result['results_summary']['compliance_score']:.1f}%") else: print(f"‚ùå Error: {response.status_code} - {response.text}")

üìà Step 4: Interpret Results

Get Detailed Results

Retrieve detailed comparison results with similarity scores and recommendations:

# Get detailed results results_url = f"https://api.nextcore.ai/v1/{{tenant_slug}}/policy-comparator/policies/{policy['id']}/results/" response = requests.get(results_url, headers=headers) if response.status_code == 200: results = response.json() for result in results: clause = result['asqa_clause_details'] print(f"\n{'='*70}") print(f"Clause: {clause['clause_number']} - {clause['title']}") print(f"Similarity Score: {result['similarity_score']:.3f}") print(f"Match Type: {result['match_type_display']}") print(f"Compliance: {'‚úì Yes' if result['is_compliant'] else '‚úó No'}") if result['matched_text']: print(f"\nMatched Text:") print(f" {result['matched_text'][:200]}...") if result['keywords_missing']: print(f"\nMissing Keywords: {', '.join(result['keywords_missing'])}") if result['recommendations']: print(f"\nRecommendations:") for rec in result['recommendations']: print(f" ‚Ä¢ {rec}")

Understanding Similarity Scores

Score Range Match Type Interpretation Recommended Action
0.8 - 1.0 Full Match Policy fully addresses ASQA requirements with clear evidence No action required. Policy is compliant.
0.6 - 0.8 Partial Match Policy covers most requirements but may need additional detail Review and consider adding more specific evidence or procedures
0.4 - 0.6 Weak Match Policy has weak coverage with missing key concepts Update policy to include missing requirements and keywords
0.0 - 0.4 No Match Policy does not adequately address this ASQA clause Create new policy section or significantly revise existing content

Gap Analysis

Get a comprehensive gap analysis with severity classification:

# Get gap analysis gap_url = f"https://api.nextcore.ai/v1/{{tenant_slug}}/policy-comparator/policies/{policy['id']}/gap_analysis/" response = requests.get(gap_url, headers=headers) if response.status_code == 200: gaps = response.json() print(f"\nüéØ Gap Analysis Summary") print(f"{'='*70}") print(f"Total Gaps: {gaps['summary']['total_gaps']}") print(f" üî¥ Critical: {gaps['summary']['critical_gaps']}") print(f" üü° Moderate: {gaps['summary']['moderate_gaps']}") print(f" üü¢ Minor: {gaps['summary']['minor_gaps']}") print(f"\nOverall Compliance Score: {gaps['summary']['compliance_score']:.1f}%") # Show critical gaps if gaps['critical_gaps']: print(f"\nüî¥ Critical Gaps Requiring Immediate Attention:") for gap in gaps['critical_gaps']: clause = gap['clause'] print(f"\n Clause {clause['clause_number']}: {clause['title']}") print(f" Similarity: {gap['similarity_score']:.3f}") print(f" Missing: {', '.join(gap['missing_keywords'][:3])}")

‚öôÔ∏è Threshold Configuration

You can customize similarity thresholds to match your organization's compliance requirements. Thresholds are configured in Django settings:

# In control_plane/settings.py or environment-specific settings POLICY_COMPARATOR = { # More strict thresholds (higher compliance bar) 'THRESHOLD_FULL_MATCH': 0.85, # Require 85% similarity for full match 'THRESHOLD_PARTIAL_MATCH': 0.70, # Partial match at 70% 'THRESHOLD_WEAK_MATCH': 0.50, # Weak match at 50% # Or more lenient thresholds (for initial assessments) # 'THRESHOLD_FULL_MATCH': 0.75, # 'THRESHOLD_PARTIAL_MATCH': 0.55, # 'THRESHOLD_WEAK_MATCH': 0.35, }

üìä Threshold Guidelines

  • Strict (0.85/0.70/0.50): Use for audit preparation or high-stakes compliance verification
  • Standard (0.80/0.60/0.40): Default balanced approach for regular compliance checking
  • Lenient (0.75/0.55/0.35): Use for initial policy development or exploratory analysis

Note: Threshold changes affect how similarity scores are classified into match types. Existing comparison results maintain their original scores but may be reclassified when viewed.
