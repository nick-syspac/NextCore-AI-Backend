Workflow 1: Evidence Upload & Verification

Systematically upload and verify compliance evidence documents.

Step 1: Prepare Evidence Documents

Actions:

  • Gather all relevant compliance documents (policies, procedures, records, certificates)
  • Organize files with clear naming conventions (e.g., "Assessment-Policy-2024.pdf")
  • Ensure documents are in supported formats (PDF, DOCX, TXT preferred)
  • Assign unique evidence numbers (e.g., "POL-001", "PROC-002", "REC-003")

Step 2: Upload Evidence (Single Document)

import requests url = "https://api.example.com/api/v1/tenants/123/audit-assistant/evidence/upload/" headers = {"Authorization": "Bearer your_jwt_token"} files = {"file": open("Assessment-Policy-2024.pdf", "rb")} data = { "evidence_number": "POL-001", "title": "Assessment Policy 2024", "description": "Comprehensive assessment policy covering validation, marking, and moderation", "evidence_type": "policy", "evidence_date": "2024-01-15", "tags": '["assessment", "policy", "validation", "2024"]' } response = requests.post(url, headers=headers, files=files, data=data) print(response.json()) # Output: {"id": 1, "status": "processing", "message": "Evidence uploaded successfully"}

Step 3: Wait for NER Processing

Processing Time: 2-5 seconds for typical documents (10-50 pages)

Status Changes: uploaded ‚Üí processing ‚Üí tagged

Step 4: Review Auto-Tagged Clauses

# Check tagged clauses response = requests.get( f"{url}/evidence/1/tagged_clauses/", headers=headers ) tagged_data = response.json() print(f"Total tagged clauses: {tagged_data['total_tagged_clauses']}") for clause in tagged_data['tagged_clauses']: print(f" - Clause {clause['asqa_clause']['clause_number']}: {clause['confidence_score']:.2f}") print(f" Mapping type: {clause['mapping_type']}") print(f" Matched keywords: {', '.join(clause['matched_keywords'][:5])}")

Step 5: Verify High-Confidence Mappings

# Verify mappings with confidence >= 0.7 high_confidence_ids = [ clause['id'] for clause in tagged_data['tagged_clauses'] if clause['confidence_score'] >= 0.7 ] response = requests.post( f"{url}/evidence/1/verify_tagging/", headers=headers, json={ "clause_evidence_ids": high_confidence_ids, "action": "verify" } ) print(response.json()) # Output: {"message": "8 clause mappings verified successfully", "verified_count": 8}

Step 6: Bulk Upload (Multiple Documents)

import os from pathlib import Path evidence_dir = Path("./evidence_documents") evidence_files = [ {"file": "Assessment-Policy-2024.pdf", "type": "policy", "number": "POL-001"}, {"file": "Assessment-Procedure-2024.docx", "type": "procedure", "number": "PROC-001"}, {"file": "Trainer-Qualifications.xlsx", "type": "record", "number": "REC-001"}, {"file": "RTO-Certificate.pdf", "type": "certificate", "number": "CERT-001"}, ] uploaded_ids = [] for evidence in evidence_files: file_path = evidence_dir / evidence["file"] files = {"file": open(file_path, "rb")} data = { "evidence_number": evidence["number"], "title": evidence["file"].replace("-", " ").replace(".pdf", "").replace(".docx", ""), "evidence_type": evidence["type"], "evidence_date": "2024-01-15" } response = requests.post(url, headers=headers, files=files, data=data) result = response.json() uploaded_ids.append(result["id"]) print(f"‚úì Uploaded: {evidence['file']} (ID: {result['id']})") print(f"\nTotal uploaded: {len(uploaded_ids)} documents")

‚úÖ Workflow Outcome

Evidence documents uploaded, NER-processed, auto-tagged with clauses, and high-confidence mappings verified.

Workflow 2: NER Processing & Review

Review and refine NER-detected entities and auto-tagged clause mappings.

Step 1: Review NER Entities

# Get evidence with NER entities response = requests.get(f"{url}/evidence/1/", headers=headers) evidence = response.json() print(f"Evidence: {evidence['title']}") print(f"NER Entities detected: {len(evidence['ner_entities'])}") print("\nEntity breakdown:") entity_types = {} for entity in evidence['ner_entities']: entity_type = entity['type'] entity_types[entity_type] = entity_types.get(entity_type, 0) + 1 for entity_type, count in entity_types.items(): print(f" - {entity_type}: {count}") # Output: # Evidence: Assessment Policy 2024 # NER Entities detected: 24 # Entity breakdown: # - STANDARD: 5 # - CLAUSE: 12 # - QUALIFICATION: 3 # - DATE: 4

Step 2: Review Low-Confidence Mappings

# Get tagged clauses response = requests.get(f"{url}/evidence/1/tagged_clauses/", headers=headers) tagged_data = response.json() # Identify low-confidence mappings for manual review low_confidence = [ clause for clause in tagged_data['tagged_clauses'] if 0.4 <= clause['confidence_score'] < 0.7 ] print(f"Low-confidence mappings requiring review: {len(low_confidence)}") for clause in low_confidence: print(f"\n Clause {clause['asqa_clause']['clause_number']}: {clause['asqa_clause']['title']}") print(f" Confidence: {clause['confidence_score']:.2f}") print(f" Strategy: {clause['mapping_type']}") print(f" Keywords matched: {', '.join(clause['matched_keywords'][:3])}")

Step 3: Manually Verify or Reject Mappings

# After manual review, verify relevant mappings verify_ids = [45, 47, 52] # Relevant clause mappings reject_ids = [48, 51] # Irrelevant mappings # Verify requests.post( f"{url}/evidence/1/verify_tagging/", headers=headers, json={"clause_evidence_ids": verify_ids, "action": "verify"} ) # Reject (delete) for reject_id in reject_ids: requests.delete(f"{url}/clause-evidence/{reject_id}/", headers=headers) print(f"‚úì Verified {len(verify_ids)} mappings") print(f"‚úó Rejected {len(reject_ids)} mappings")

Step 4: Add Manual Mappings

# Add manual mapping for clause not auto-detected response = requests.post( f"{url}/clause-evidence/", headers=headers, json={ "asqa_clause": 28, # Clause 1.7.2 "evidence": 1, "mapping_type": "manual", "relevance_notes": "Section 4.2 of policy addresses continuous improvement requirements" } ) print("‚úì Manual mapping created")

Step 5: Re-process Evidence (if needed)

# Trigger manual NER re-processing (e.g., if auto-processing failed) response = requests.post( f"{url}/evidence/1/process_ner/", headers=headers ) task_id = response.json()['task_id'] print(f"NER processing re-started: {task_id}") # Check status after 5 seconds import time time.sleep(5) response = requests.get(f"{url}/evidence/1/", headers=headers) print(f"Status: {response.json()['status']}") # Should be 'tagged'

‚úÖ Workflow Outcome

NER entities reviewed, low-confidence mappings manually verified or rejected, manual mappings added for missed clauses.

Workflow 3: Gap Identification & Remediation

Systematically identify and address compliance gaps before audit.

Step 1: Identify All Gaps

# Get gaps for Standards 1-3 response = requests.get( f"{url}/clause-evidence/gaps/", headers=headers, params={"standard_ids": "1,2,3"} ) gap_data = response.json() print(f"Coverage: {gap_data['coverage_percentage']:.1f}%") print(f"Total clauses: {gap_data['total_clauses']}") print(f"Clauses without evidence: {gap_data['clauses_without_evidence']}") print(f"\nGaps by severity:") severity_counts = {} for gap in gap_data['gaps']: severity = gap['severity'] severity_counts[severity] = severity_counts.get(severity, 0) + 1 for severity, count in severity_counts.items(): print(f" - {severity.upper()}: {count}")

Step 2: Prioritize Critical Gaps

# Extract critical gaps for immediate action critical_gaps = [ gap for gap in gap_data['gaps'] if gap['severity'] == 'critical' ] print(f"\nüî¥ CRITICAL GAPS ({len(critical_gaps)}):") for i, gap in enumerate(critical_gaps, 1): clause = gap['clause'] print(f"\n{i}. Clause {clause['clause_number']}: {clause['title']}") print(f" Standard: {clause['standard_number']}") print(f" Recommendation: {gap['recommendation']}")

Step 3: Upload Missing Evidence

# Address critical gap: Upload marketing compliance policy files = {"file": open("Marketing-Compliance-Policy.pdf", "rb")} data = { "evidence_number": "POL-005", "title": "Marketing and Advertising Compliance Policy", "description": "Policy addressing ASQA marketing and advertising requirements", "evidence_type": "policy", "evidence_date": "2024-02-01", "tags": '["marketing", "advertising", "compliance", "policy"]' } response = requests.post(url, headers=headers, files=files, data=data) evidence_id = response.json()['id'] print(f"‚úì Uploaded marketing policy (ID: {evidence_id})") # Wait for NER processing import time time.sleep(5) # Manually map to critical clause if not auto-detected response = requests.post( f"{url}/clause-evidence/", headers=headers, json={ "asqa_clause": 25, # Clause 1.5.3 (Marketing compliance) "evidence": evidence_id, "mapping_type": "manual", "relevance_notes": "Comprehensive marketing compliance policy addressing all ASQA requirements" } ) print("‚úì Mapped to Clause 1.5.3")

Step 4: Re-check Gap Coverage

# Re-run gap analysis after uploading evidence response = requests.get( f"{url}/clause-evidence/gaps/", headers=headers, params={"standard_ids": "1,2,3"} ) updated_gap_data = response.json() print(f"\nüìä Updated Coverage:") print(f" Before: {gap_data['coverage_percentage']:.1f}%") print(f" After: {updated_gap_data['coverage_percentage']:.1f}%") print(f" Improvement: +{updated_gap_data['coverage_percentage'] - gap_data['coverage_percentage']:.1f}%") print(f"\n Remaining gaps: {updated_gap_data['clauses_without_evidence']}") # Check if critical gap was addressed remaining_critical = [ gap for gap in updated_gap_data['gaps'] if gap['severity'] == 'critical' ] print(f" Critical gaps remaining: {len(remaining_critical)}")

Step 5: Generate Gap Remediation Report

# Export gap data for stakeholder review import json remediation_report = { "report_date": "2024-04-01", "total_clauses": updated_gap_data['total_clauses'], "coverage_percentage": updated_gap_data['coverage_percentage'], "gaps_by_severity": { "critical": len([g for g in updated_gap_data['gaps'] if g['severity'] == 'critical']), "major": len([g for g in updated_gap_data['gaps'] if g['severity'] == 'major']), "minor": len([g for g in updated_gap_data['gaps'] if g['severity'] == 'minor']) }, "gaps": [ { "clause": f"{g['clause']['clause_number']}: {g['clause']['title']}", "severity": g['severity'], "recommendation": g['recommendation'] } for g in updated_gap_data['gaps'] ] } with open('gap_remediation_report.json', 'w') as f: json.dump(remediation_report, f, indent=2) print("‚úì Remediation report exported to gap_remediation_report.json")

‚úÖ Workflow Outcome

All compliance gaps identified, prioritized by severity, critical gaps addressed with evidence uploads, and remediation report generated.

Step 2: Review Detailed Clause Entries

# Get detailed report with all clause entries response = requests.get(f"{url}/audit-reports/{report_id}/", headers=headers) detailed_report = response.json() print(f"\nüìã Clause-by-Clause Breakdown:") print(f"Total entries: {len(detailed_report['clause_entries'])}") # Group by compliance status status_groups = {} for entry in detailed_report['clause_entries']: status = entry['compliance_status'] status_groups[status] = status_groups.get(status, 0) + 1 for status, count in status_groups.items(): print(f" - {status.replace('_', ' ').title()}: {count}") # Show non-compliant clauses non_compliant = [ entry for entry in detailed_report['clause_entries'] if entry['compliance_status'] == 'non_compliant' ] print(f"\nüî¥ Non-Compliant Clauses ({len(non_compliant)}):") for entry in non_compliant[:5]: # Show first 5 clause = entry['asqa_clause'] print(f" - {clause['clause_number']}: {clause['title']}") print(f" Finding: {entry['finding']}") print(f" Severity: {entry['severity']}")

Step 3: Add Findings and Recommendations

# Update report with auditor findings findings = [] recommendations = [] for entry in non_compliant: clause_num = entry['asqa_clause']['clause_number'] finding = { "clause": clause_num, "finding": entry['finding'], "severity": entry['severity'], "recommendation": entry['recommendation'] } findings.append(finding) # Add overall recommendations recommendations = [ "Address all critical gaps before ASQA submission", "Verify all auto-tagged evidence with compliance officer", "Upload missing trainer qualification certificates for Standard 2", "Complete financial viability documentation for Standard 3", "Implement continuous improvement tracking system" ] # Update report response = requests.patch( f"{url}/audit-reports/{report_id}/", headers=headers, json={ "findings": findings, "recommendations": recommendations, "status": "review" } ) print("‚úì Report updated with findings and recommendations")

Workflow 5: Continuous Compliance Monitoring

Maintain ongoing compliance with regular evidence updates and gap monitoring.

Quarterly Compliance Review Process

Quarter Start:

  1. Generate gap analysis for all standards
  2. Identify new or persistent gaps
  3. Create action plan for gap remediation
  4. Assign responsibilities to compliance team

Monthly Activities:

  1. Upload new evidence documents (policies, procedures, records)
  2. Review and verify auto-tagged clause mappings
  3. Address gaps identified in previous month
  4. Generate compliance summary report

Quarter End:

  1. Generate comprehensive audit report
  2. Review compliance percentage trends
  3. Document improvements and remaining gaps
  4. Plan next quarter activities

Automated Monitoring Script

from datetime import datetime, timedelta def quarterly_compliance_check(tenant_id, jwt_token): """Run quarterly compliance check""" base_url = f"https://api.example.com/api/v1/tenants/{tenant_id}/audit-assistant" headers = {"Authorization": f"Bearer {jwt_token}"} # Get current quarter dates today = datetime.now() quarter_start = datetime(today.year, ((today.month - 1) // 3) * 3 + 1, 1) quarter_end = quarter_start + timedelta(days=90) print(f"Quarterly Compliance Check: {quarter_start.date()} to {quarter_end.date()}") # 1. Identify gaps response = requests.get( f"{base_url}/clause-evidence/gaps/", headers=headers, params={"standard_ids": "1,2,3,4,5,6,7,8"} ) gap_data = response.json() print(f"\nüìä Compliance Status:") print(f" Coverage: {gap_data['coverage_percentage']:.1f}%") print(f" Gaps: {gap_data['clauses_without_evidence']}") # 2. Count critical gaps critical_gaps = [g for g in gap_data['gaps'] if g['severity'] == 'critical'] print(f" Critical gaps: {len(critical_gaps)}") # 3. Generate quarterly report response = requests.post( f"{base_url}/audit-reports/generate_report/", headers=headers, json={ "report_number": f"AR-{today.year}-Q{(today.month - 1) // 3 + 1}", "title": f"Q{(today.month - 1) // 3 + 1} {today.year} Compliance Report", "asqa_standards": list(range(1, 9)), "audit_period_start": quarter_start.date().isoformat(), "audit_period_end": quarter_end.date().isoformat() } ) report = response.json() print(f"\n‚úì Quarterly report generated (ID: {report['id']})") print(f" Overall compliance: {report['compliance_percentage']:.1f}%") print(f" Critical compliance: {report['critical_compliance_percentage']:.1f}%") return { "coverage": gap_data['coverage_percentage'], "critical_gaps": len(critical_gaps), "report_id": report['id'] } # Run check result = quarterly_compliance_check(123, "your_jwt_token")

‚úÖ Workflow Outcome

Continuous compliance monitoring established with quarterly reviews, monthly evidence updates, and automated gap tracking.

Workflow 6: Multi-User Collaboration

Coordinate compliance activities across multiple team members.

Role-Based Workflow

Compliance Officer:

  • Generate gap analysis reports
  • Assign evidence collection tasks to department heads
  • Review and verify all clause mappings
  • Approve evidence before inclusion in audit reports
  • Submit final reports to ASQA

Department Heads:

  • Upload department-specific evidence (policies, procedures)
  • Add relevant tags for categorization
  • Provide descriptions and context for evidence
  • Respond to evidence requests from compliance officer

Auditors (External/Internal):

  • Review audit reports and clause entries
  • Add findings and recommendations
  • Request additional evidence for specific clauses
  • Validate evidence quality and relevance
