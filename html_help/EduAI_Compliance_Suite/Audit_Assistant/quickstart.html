Step 1: Upload Evidence Document

Upload your first evidence document with auto-tagging enabled. The system will extract text and automatically identify relevant ASQA clauses.

Using Python

import requests from datetime import date # Configuration BASE_URL = "https://api.nextcore.ai/v1/{tenant_slug}/audit-assistant" HEADERS = {"Authorization": "Bearer YOUR_TOKEN"} # Prepare evidence file evidence_file = open("Assessment_Validation_Policy.pdf", "rb") # Upload with auto-tagging upload_data = { "evidence_number": "EV-001", "title": "Assessment Validation Policy", "description": "RTO policy for assessment validation and moderation", "evidence_type": "policy", "evidence_date": date.today().isoformat(), "tags": ["assessment", "validation", "standard-1"], "auto_tag": "true" # Enable NER auto-tagging } files = {"file": evidence_file} response = requests.post( f"{BASE_URL}/evidence/upload/", data=upload_data, files=files, headers=HEADERS ) evidence = response.json() print(f"‚úÖ Evidence uploaded: {evidence['id']}") print(f" Status: {evidence['status']}") print(f" Processing: {evidence.get('processing', False)}") # Close file evidence_file.close()

Using cURL

curl -X POST "https://api.nextcore.ai/v1/{tenant_slug}/audit-assistant/evidence/upload/" \ -H "Authorization: Bearer YOUR_TOKEN" \ -F "file=@Assessment_Validation_Policy.pdf" \ -F "evidence_number=EV-001" \ -F "title=Assessment Validation Policy" \ -F "description=RTO policy for assessment validation and moderation" \ -F "evidence_type=policy" \ -F "evidence_date=2025-11-04" \ -F "tags=[\"assessment\",\"validation\"]" \ -F "auto_tag=true"

Expected Response

{ "id": 42, "evidence_number": "EV-001", "title": "Assessment Validation Policy", "description": "RTO policy for assessment validation and moderation", "evidence_type": "policy", "file_url": "https://storage.example.com/evidence/tenant_1/abc123.pdf", "file_name": "abc123.pdf", "file_size": 245760, "status": "processing", "tags": ["assessment", "validation", "standard-1"], "evidence_date": "2025-11-04", "uploaded_at": "2025-11-04T10:30:00Z", "uploaded_by": 5, "uploaded_by_name": "John Smith", "ner_entities": null, "ner_processed_at": null, "processing": true }

üí° What Happens During Processing?

  1. Text Extraction: Extract text from PDF/DOCX/TXT file
  2. NER Detection: Identify standards, clauses, qualifications, dates, orgs, policies
  3. Clause Matching: Match document to ASQA clauses using 4 strategies
  4. Confidence Scoring: Calculate confidence scores (0.4-1.0) for each mapping
  5. Status Update: Change status from "processing" ‚Üí "tagged"

Processing typically completes in 2-5 seconds.

Step 2: Review Auto-Tagged Clauses

After processing completes, review the clauses that were automatically tagged to your evidence.

# Get tagged clauses for evidence evidence_id = 42 response = requests.get( f"{BASE_URL}/evidence/{evidence_id}/tagged_clauses/", headers=HEADERS ) tagged_clauses = response.json() print(f"\nüìã Auto-Tagged Clauses: {len(tagged_clauses)}\n") for mapping in tagged_clauses: clause = mapping['clause_number'] title = mapping['clause_title'] confidence = mapping['confidence_score'] mapping_type = mapping['mapping_type_display'] verified = mapping['is_verified'] print(f"Clause {clause}: {title}") print(f" Confidence: {confidence:.2f} ({mapping_type})") print(f" Verified: {'‚úÖ' if verified else '‚è≥'}") if mapping['matched_keywords']: print(f" Keywords: {', '.join(mapping['matched_keywords'][:5])}") if mapping['matched_entities']: entities = [e['entity'] for e in mapping['matched_entities'][:3]] print(f" Entities: {', '.join(entities)}") print()

Step 3: Verify Mappings

Verify auto-tagged mappings that are accurate. High-confidence mappings (>0.8) are typically reliable, but always review suggested matches (<0.6).

# Verify specific clause mappings clause_evidence_ids = [101, 102, 103] # IDs from tagged_clauses response verify_response = requests.post( f"{BASE_URL}/evidence/{evidence_id}/verify_tagging/", json={"clause_evidence_ids": clause_evidence_ids}, headers=HEADERS ) result = verify_response.json() print(result['message']) # Output: "3 clause mappings verified." # Evidence status automatically updates to "reviewed" updated_evidence = result['evidence'] print(f"Evidence status: {updated_evidence['status']}") # Output: "Evidence status: reviewed"

‚ö†Ô∏è Review Guidelines

  • Confidence ‚â•0.8: Usually accurate - quick verification OK
  • Confidence 0.6-0.8: Review document context before verifying
  • Confidence <0.6: Suggested only - careful review required
  • Missing Clauses: Manually add mappings for clauses not auto-detected

Step 4: Create Audit Report

Generate a comprehensive audit report covering selected ASQA standards.

from datetime import date, timedelta # Get standard IDs (typically Standards 1-8) standards_response = requests.get( f"https://api.nextcore.ai/v1/{tenant_slug}/policy-comparator/standards/", headers=HEADERS ) standards = standards_response.json()['results'] standard_ids = [s['id'] for s in standards[:3]] # Standards 1-3 for this example # Create audit report audit_data = { "report_number": "AUDIT-2025-Q4", "title": "Q4 2025 Compliance Audit", "description": "Quarterly compliance review covering Standards 1-3", "asqa_standards": standard_ids, "audit_period_start": (date.today() - timedelta(days=90)).isoformat(), "audit_period_end": date.today().isoformat(), "status": "in_progress" } response = requests.post( f"{BASE_URL}/audit-reports/", json=audit_data, headers=HEADERS ) audit_report = response.json() print(f"\n‚úÖ Audit Report Created: {audit_report['report_number']}") print(f" Total Clauses: {audit_report['total_clauses']}") print(f" Clauses with Evidence: {audit_report['clauses_with_evidence']}") print(f" Compliance: {audit_report['compliance_percentage']:.1f}%") print(f" Critical Compliance: {audit_report['critical_compliance_percentage']:.1f}%")

# Retrieve full report with clause entries report_id = audit_report['id'] detail_response = requests.get( f"{BASE_URL}/audit-reports/{report_id}/", headers=HEADERS ) detailed_report = detail_response.json() # View clause entries print(f"\nüìä Clause-by-Clause Breakdown:\n") for entry in detailed_report['clause_entries'][:5]: # First 5 clauses clause_num = entry['clause_number'] clause_title = entry['clause_title'] status = entry['compliance_status_display'] evidence_count = entry['evidence_count'] verified_count = entry['verified_evidence_count'] print(f"{clause_num}: {clause_title}") print(f" Status: {status}") print(f" Evidence: {evidence_count} total, {verified_count} verified") if entry['finding']: print(f" Finding: {entry['finding']}") print()

Step 5: Analyze Gaps

Identify clauses with insufficient evidence and get recommendations for remediation.

# Run gap analysis for specific standards gap_params = {"standard_ids": standard_ids} gap_response = requests.get( f"{BASE_URL}/clause-evidence/gaps/", params=gap_params, headers=HEADERS ) gap_analysis = gap_response.json() print(f"\nüîç Gap Analysis Results\n") print(f"Total Clauses: {gap_analysis['total_clauses']}") print(f"Clauses with Gaps: {gap_analysis['clauses_with_gaps']}") print(f" üî¥ Critical: {gap_analysis['critical_gaps']}") print(f" üü° Major: {gap_analysis['major_gaps']}") print(f" üü¢ Minor: {gap_analysis['minor_gaps']}") # Show critical gaps if gap_analysis['critical_gaps'] > 0: print(f"\nüî¥ Critical Gaps Requiring Immediate Action:\n") critical = [g for g in gap_analysis['gaps'] if g['gap_severity'] == 'critical'] for gap in critical: print(f"Clause {gap['clause_number']}: {gap['clause_title']}") print(f" Compliance Level: {gap['compliance_level']}") print(f" Current Evidence: {gap['evidence_count']}") print(f" Verified: {gap['verified_evidence_count']}") print(f"\n Recommendations:") for rec in gap['recommendations']: print(f" ‚Ä¢ {rec}") print()
